% 设置正文字号为 11pt，行距 14pt
\renewcommand{\normalsize}{\fontsize{11pt}{14pt}\selectfont}

\xeCJKsetup{
	CJKglue = \hspace{0pt}, % 防止中英文粘连
	CheckSingle = true      % 单个汉字也启用 CJK 字体
}

% Serif
\setCJKmainfont{SourceHanSerifSC}[
	Path = ./fonts/SourceHanSerifSC/,
	Extension = .otf,
	UprightFont = *-Regular,
	BoldFont = *-Bold,
	ItalicFont = *-Regular, % fallback 到 Regular
	AutoFakeSlant = 0.25    % 斜体时轻微 fake
]

\newfontfamily\avaliscratch{Avali Scratch}[

	UprightFont = avali-scratch,
